38:           # the :authenticate_on_every_request config option to false. 
39:           log.debug "Existing local CAS session detected for #{session[@client.username_session_key].inspect}. "+
40:                   "Previous ticket #{last_service_ticket.ticket.inspect} will be re-used."

42:         end
43: 
44:         if service_ticket
45:           @client.validate_service_ticket(service_ticket) unless service_ticket.has_been_validated?
